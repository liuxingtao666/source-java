		--增加管理员VO，用于封装查询结果。
		a、DAO（*）
			--追加查询方法
				包含搜索条件
				包含分页参数
				需要多表联合查询
				结果需要封装到VO
			--查询数据SQL
			select * from (
				select a.*,rownum r from admin_info a
				where id in (
					select ai.id
					from admin_info ai
					inner join admin_role ar on ai.id=ar.admin_id
					inner join role_info ri on ri.id=ar.role_id
					inner join role_privilege rp on ri.id=rp.role_id
					where 1=1
					and ri.name='皇叔'
					and rp.privilege_id=1
				)
			) where r<6 and r>0				
			--查询总行数SQL
				select count(*) from admin_info
				where id in (
					select ai.id
					from admin_info ai
					inner join admin_role ar on ai.id=ar.admin_id
					inner join role_info ri on ri.id=ar.role_id
					inner join role_privilege rp on ri.id=rp.role_id
					where 1=1
					and ri.name='皇叔'
					and rp.privilege_id=1
				)
